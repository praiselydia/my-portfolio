# Function to compute point-biserial correlation
def compute_point_biserial_correlation(dataframe, binary_col, continuous_col):
    return pointbiserialr(dataframe[binary_col], dataframe[continuous_col])[0]

# Function to compute chi-squared test for association
def compute_chi_squared_association(dataframe, binary_col, categorical_col):
    contingency_table = pd.crosstab(dataframe[binary_col], dataframe[categorical_col])
    chi2, p, dof, expected = chi2_contingency(contingency_table)
    return chi2

import matplotlib.pyplot as plt
import seaborn as sns

# Separate continuous and categorical variables
continuous_corr = {k: v for k, v in correlation_results.items() if abs(v) <= 1}  # Point-biserial correlation values
categorical_corr = {k: v for k, v in correlation_results.items() if v > 1}  # Chi-squared values

# Convert to DataFrame
continuous_df = pd.DataFrame.from_dict(continuous_corr, orient='index', columns=['Correlation'])
categorical_df = pd.DataFrame.from_dict(categorical_corr, orient='index', columns=['Chi-squared'])

# Set up the figure and axis
fig, ax1 = plt.subplots(figsize=(14, 8))

# Plot continuous variables
color = 'tab:blue'
ax1.set_xlabel('Variables')
ax1.set_ylabel('Point-biserial Correlation', color=color)
ax1.bar(continuous_df.index, continuous_df['Correlation'], color=color, alpha=0.6, label='Continuous Variables')
ax1.tick_params(axis='y', labelcolor=color)
plt.xticks(rotation=45, ha='right')

# Create a second y-axis for categorical variables
ax2 = ax1.twinx()  
color = 'tab:green'
ax2.set_ylabel('Chi-squared Value', color=color)  
ax2.bar(categorical_df.index, categorical_df['Chi-squared'], color=color, alpha=0.6, label='Categorical Variables')
ax2.tick_params(axis='y', labelcolor=color)

# Add a horizontal line at y=0 for the first axis
ax1.axhline(0, color='black', linewidth=0.8)

# Add legends
fig.legend(loc='upper right', bbox_to_anchor=(1, 1), bbox_transform=ax1.transAxes)

# Set plot title and layout
plt.title('Correlation and Association of Variables with Churn')
fig.tight_layout()  

# Show the plot
plt.show()

From the correlation found between the variables, we could see that there is a high positive correlation between complain and churn. Which means as much as the customer complains, he leaves the company. It is crucial to analyze the sentiments of customers through the complain to identify what the customer is trying to say. By maintaining the level of grievance displayed by the customer, and its correlation with exiting of previous customers, we can identify if the customer aims to leave the company and can provide him with ample compensation to make him/her stay with the company.
